35: class Instagram {
36:     /**
37:      * Gets a media from Instagram
38:      * @param string $id Media ID
39:      * @return array
40:      * @uses MeTools\Utility\Xml::fromFile()
41:      */
42:     public static function getMedia($id) {
43:         //See https://www.instagram.com/developer/endpoints/media/#get_media
44:         $url = 'https://api.instagram.com/v1/media/%s?access_token=%s';
45:         $photo = Xml::fromFile(sprintf($url, $id, config('Instagram.key')));
46:         
47:         return (object) [
48:             'path' => $photo['data']['images']['standard_resolution']['url']
49:         ];
50:     }
51: 
52:     /**
53:      * Gets the recent media for an user from Instagram
54:      * @return array
55:      * @uses MeTools\Utility\Xml::fromFile()
56:      */
57:     public static function getRecentUser() {
58:         //See https://www.instagram.com/developer/endpoints/users/#get_users_media_recent_self
59:         $url = 'https://api.instagram.com/v1/users/self/media/recent/?access_token=%s';
60:         $photos = Xml::fromFile(sprintf($url, config('Instagram.key')));
61:         
62:         return array_map(function($photo) {
63:             return (object) [
64:                 'id'            => $photo['id'],
65:                 'description'   => $photo['caption']['text'],
66:                 'path'          => $photo['images']['standard_resolution']['url']
67:             ];
68:         }, $photos['data']);
69:     }
